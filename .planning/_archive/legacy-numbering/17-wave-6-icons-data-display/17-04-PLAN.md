  <behavior>
    - `<nav aria-label={ariaLabel}>` wraps `<ol className="ds-atom-breadcrumbs">`.
    - Each item is `<li>`. If `item.href` provided, render `<a href={href} onClick={onClick}>{label}</a>`. Else render `<span>{label}</span>`.
    - Last item has `aria-current="page"` on the anchor or span.
    - Separator (ChevronRight from canonical icons subpath, size 14) between items, NOT after the last.
    - When `items.length > maxVisible` (default 4): show first item, then a "more" trigger button (with MoreHorizontal icon), then the last `maxVisible - 2` items. The middle hidden items render inside a DSDropdown opened on click.
    - DSDropdown role="menu" with role="menuitem" entries (NOT listbox/option — this is navigation, not selection).
    - Dropdown items closing menu on selection: clicking an item navigates (calls onClick or follows href) and closes dropdown.
    - On narrow widths the consumer can pass smaller maxVisible (3 or 2). Below 2 is unsupported (always show first + last at minimum).
    - Dark-mode: ink-2 for inactive items, ink for current page.

    ```tsx
    /**
     * # Usage Audit — Breadcrumbs (DS-69)
     *
     * Consumers (post v0.6):
     * - JobDash app shell — current-page hierarchical nav
     * - Settings pages — section path
     *
     * API:
     * - items: BreadcrumbItem[] (label + optional href + onClick)
     * - maxVisible: number (default 4 — show first + trigger + last (maxVisible-2))
     * - ariaLabel: string (default "Breadcrumb")
     *
     * Implementation:
     * - <nav><ol> with aria-current="page" on last item per WAI-ARIA
     * - Truncation: middle items collapse into a DSDropdown trigger button
     * - ChevronRight separator from canonical /icons subpath
     */
    import { forwardRef, useRef, useState } from "react";
    import { ChevronRight, MoreHorizontal } from "./icons";
    import { DSDropdown } from "./_internals/DSDropdown";

    export interface BreadcrumbItem { ... }
    export interface BreadcrumbsProps { ... }

    export const Breadcrumbs = forwardRef<HTMLElement, BreadcrumbsProps>(function Breadcrumbs(
      { items, maxVisible = 4, ariaLabel = "Breadcrumb", className, style }, ref,
    ) {
      const [overflowOpen, setOverflowOpen] = useState(false);
      const moreBtnRef = useRef<HTMLButtonElement | null>(null);

      const shouldTruncate = items.length > maxVisible && maxVisible >= 3;

      // Layout when truncated:
      //   [items[0]] [/] [more-trigger] [/] [items[items.length - (maxVisible-2)]] ... [items[last]]
      let visible: { kind: "item" | "more"; data?: BreadcrumbItem | BreadcrumbItem[]; index?: number }[];
      if (!shouldTruncate) {
        visible = items.map((item, i) => ({ kind: "item", data: item, index: i }));
      } else {
        const first = items[0]!;
        const tailStart = items.length - (maxVisible - 2);
        const hidden = items.slice(1, tailStart);
        const tail = items.slice(tailStart);
        visible = [
          { kind: "item", data: first, index: 0 },
          { kind: "more", data: hidden },
          ...tail.map((item, i) => ({ kind: "item" as const, data: item, index: tailStart + i })),
        ];
      }

      const handleSelectHidden = (i: number, hidden: BreadcrumbItem[]) => {
        const item = hidden[i]!;
        if (item.onClick) item.onClick({} as React.MouseEvent);
        else if (item.href) window.location.href = item.href;
        setOverflowOpen(false);
      };

      return (
        <nav ref={ref} aria-label={ariaLabel} className={`ds-atom-breadcrumbs${className ? ` ${className}` : ""}`} style={style}>
          <ol className="ds-atom-breadcrumbs-list">
            {visible.map((entry, idx) => {
              const isLastItem = idx === visible.length - 1;
              if (entry.kind === "more") {
                const hidden = entry.data as BreadcrumbItem[];
                return (
                  <li key="__more__" className="ds-atom-breadcrumbs-li">
                    <button
                      ref={moreBtnRef}
                      type="button"
                      className="ds-atom-breadcrumbs-more"
                      aria-label={`Show ${hidden.length} hidden breadcrumb${hidden.length === 1 ? "" : "s"}`}
                      aria-expanded={overflowOpen}
                      aria-haspopup="menu"
                      onClick={() => setOverflowOpen(o => !o)}
                    >
                      <MoreHorizontal size={14} />
                    </button>
                    {!isLastItem && <ChevronRight size={14} className="ds-atom-breadcrumbs-sep" />}
                    <DSDropdown
                      anchorRef={moreBtnRef}
                      open={overflowOpen}
                      onOpenChange={setOverflowOpen}
                      activeIndex={0}
                      onActiveIndexChange={() => {}}
                      itemCount={hidden.length}
                      onSelect={(i) => handleSelectHidden(i, hidden)}
                      typeAheadGetText={(i) => hidden[i]?.label ?? ""}
                    >
                      <ul role="menu" className="ds-atom-breadcrumbs-menu">
                        {hidden.map((h, i) => (
                          <li key={i} role="menuitem" className="ds-atom-breadcrumbs-menuitem">
                            {h.href ? (
                              <a href={h.href} onClick={(e) => { h.onClick?.(e); setOverflowOpen(false); }}>
                                {h.label}
                              </a>
                            ) : (
                              <button type="button" onClick={() => { h.onClick?.({} as React.MouseEvent); setOverflowOpen(false); }}>
                                {h.label}
                              </button>
                            )}
                          </li>
                        ))}
                      </ul>
                    </DSDropdown>
                  </li>
                );
              }
              const item = entry.data as BreadcrumbItem;
              const isCurrent = entry.index === items.length - 1;
              return (
                <li key={entry.index} className="ds-atom-breadcrumbs-li">
                  {item.href ? (
                    <a
                      href={item.href}
                      onClick={item.onClick}
                      aria-current={isCurrent ? "page" : undefined}
                      className={isCurrent ? "ds-atom-breadcrumbs-current" : "ds-atom-breadcrumbs-link"}
                    >{item.label}</a>
                  ) : (
                    <span
                      aria-current={isCurrent ? "page" : undefined}
                      className={isCurrent ? "ds-atom-breadcrumbs-current" : "ds-atom-breadcrumbs-text"}
                    >{item.label}</span>
                  )}
                  {!isLastItem && <ChevronRight size={14} className="ds-atom-breadcrumbs-sep" />}
                </li>
              );
            })}
          </ol>
        </nav>
      );
    });
    ```
