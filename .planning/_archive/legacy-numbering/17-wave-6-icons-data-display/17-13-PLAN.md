    ```tsx
    /**
     * # Usage Audit — RichText (DS-70, D-17-14..D-17-19)
     *
     * Built on @tiptap/react 3.x + StarterKit + Link + Placeholder.
     * Heaviest primitive in Phase 17 (~50-70 KB gzipped TipTap deps).
     * TipTap packages are externalized in tsup.config.ts (Plan 00).
     *
     * ## Controlled-value sync (DO NOT BREAK)
     *
     * TipTap is NOT natively a controlled component. The naive pattern infinite-loops:
     *   useEffect(() => editor.setContent(value), [value]); // BAD
     * because setContent fires onUpdate → onChange → parent setState → new value → effect → loop.
     *
     * Three-layer guard (RESEARCH.md § "Controlled Value Sync"):
     *   1. lastEmittedRef tracks the last value we emitted.
     *   2. Compare value with editor.getHTML() before calling setContent (defensive equality).
     *   3. Pass { emitUpdate: false } to setContent so it doesn't fire onUpdate.
     *
     * ## Output formats (D-17-17)
     *
     * - "html" (default): onChange(editor.getHTML())  — string
     * - "json":           onChange(editor.getJSON())  — TipTap Doc object
     *
     * ## Sanitization (D-17-18)
     *
     * StarterKit's extension allowlist filters tags/attrs on paste — equivalent to schema-based
     * sanitization. NO DOMPurify dep. Server-side sanitization is the consumer's responsibility.
     *
     * ## Threat model
     *
     * - XSS via paste: TipTap allowlist mitigates (T-17-13-01)
     * - XSS via consumer-supplied initial value: caller's responsibility (we render via TipTap which
     *   parses + sanitizes during initial setContent)
     */
    import { forwardRef, useEffect, useRef, useState } from "react";
    import { EditorContent, useEditor } from "@tiptap/react";
    import StarterKit from "@tiptap/starter-kit";
    import Link from "@tiptap/extension-link";
    import Placeholder from "@tiptap/extension-placeholder";
    import UnderlineExtension from "@tiptap/extension-underline";
    import {
      Bold, Italic, Underline, Strikethrough, Code, List, ListOrdered,
      Heading2, Heading3, Quote, Minus, Link2, ChevronDown,
    } from "./icons";
    import { Button } from "./Button";
    import { DSDropdown } from "./_internals/DSDropdown";
    import { DSPortal } from "./_internals/DSPortal";

    export interface RichTextProps {
      value: string | object;
      onChange: (value: string | object) => void;
      placeholder?: string;
      readOnly?: boolean;
      outputFormat?: "html" | "json";
      toolbar?: React.ReactNode;
      className?: string;
      ariaLabel?: string;
      style?: React.CSSProperties;
    }

    export const RichText = forwardRef<HTMLDivElement, RichTextProps>(function RichText(
      { value, onChange, placeholder, readOnly = false, outputFormat = "html", toolbar, className, ariaLabel = "Rich text editor", style },
      ref,
    ) {
      const lastEmittedRef = useRef<string>(typeof value === "string" ? value : "");
      const linkBtnRef = useRef<HTMLButtonElement | null>(null);
      const headingBtnRef = useRef<HTMLButtonElement | null>(null);
      const [linkOpen, setLinkOpen] = useState(false);
      const [headingOpen, setHeadingOpen] = useState(false);
      const [headingActiveIndex, setHeadingActiveIndex] = useState(0);
      const [linkUrl, setLinkUrl] = useState("");

      const editor = useEditor({
        extensions: [
          StarterKit,
          Link.configure({ openOnClick: false, autolink: true }),
          Placeholder.configure({ placeholder: placeholder ?? "" }),
          UnderlineExtension,
        ],
        content: value,
        editable: !readOnly,
        immediatelyRender: false,
        onUpdate: ({ editor }) => {
          if (outputFormat === "json") {
            onChange(editor.getJSON());
          } else {
            const html = editor.getHTML();
            lastEmittedRef.current = html;
            onChange(html);
          }
        },
      });

      // Controlled value → editor sync (the trap — see header docblock)
      useEffect(() => {
        if (!editor) return;
        if (typeof value !== "string") return;
        if (value === lastEmittedRef.current) return;
        if (value === editor.getHTML()) return;
        editor.commands.setContent(value, { emitUpdate: false });
      }, [editor, value]);

      // editable toggle
      useEffect(() => {
        if (!editor) return;
        if (editor.isEditable !== !readOnly) {
          editor.setEditable(!readOnly);
        }
      }, [editor, readOnly]);

      const headingItems = [
        { label: "Paragraph", action: () => editor?.chain().focus().setParagraph().run() },
        { label: "Heading 2", action: () => editor?.chain().focus().toggleHeading({ level: 2 }).run() },
        { label: "Heading 3", action: () => editor?.chain().focus().toggleHeading({ level: 3 }).run() },
      ];

      const isActive = (n: string, attrs?: Record<string, unknown>) => editor?.isActive(n, attrs) ?? false;

      const defaultToolbar = (
        <div role="toolbar" aria-label="Formatting" className="ds-atom-richtext-toolbar">
          <Button variant="ghost" size="sm" aria-label="Bold" aria-pressed={isActive("bold")} data-active={isActive("bold") || undefined} onClick={() => editor?.chain().focus().toggleBold().run()}><Bold size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Italic" aria-pressed={isActive("italic")} data-active={isActive("italic") || undefined} onClick={() => editor?.chain().focus().toggleItalic().run()}><Italic size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Underline" aria-pressed={isActive("underline")} data-active={isActive("underline") || undefined} onClick={() => editor?.chain().focus().toggleUnderline().run()}><Underline size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Strikethrough" aria-pressed={isActive("strike")} data-active={isActive("strike") || undefined} onClick={() => editor?.chain().focus().toggleStrike().run()}><Strikethrough size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Inline code" aria-pressed={isActive("code")} data-active={isActive("code") || undefined} onClick={() => editor?.chain().focus().toggleCode().run()}><Code size={16} /></Button>
          <span className="ds-atom-richtext-toolbar-divider" aria-hidden="true" />
          <Button ref={headingBtnRef} variant="ghost" size="sm" aria-label="Heading style" aria-haspopup="menu" aria-expanded={headingOpen} onClick={() => setHeadingOpen(o => !o)} data-active={isActive("heading") || undefined}>
            <Heading2 size={16} /><ChevronDown size={12} />
          </Button>
          <DSDropdown
            anchorRef={headingBtnRef}
            open={headingOpen}
            onOpenChange={setHeadingOpen}
            activeIndex={headingActiveIndex}
            onActiveIndexChange={setHeadingActiveIndex}
            itemCount={headingItems.length}
            onSelect={(i) => { headingItems[i]?.action(); setHeadingOpen(false); }}
            typeAheadGetText={(i) => headingItems[i]?.label ?? ""}
          >
            <ul role="menu" className="ds-atom-richtext-headingmenu">
              {headingItems.map((it, i) => (
                <li key={it.label} role="menuitem" data-active={headingActiveIndex === i || undefined}>
                  <button type="button" onClick={() => { it.action(); setHeadingOpen(false); }}>{it.label}</button>
                </li>
              ))}
            </ul>
          </DSDropdown>
          <Button variant="ghost" size="sm" aria-label="Bulleted list" aria-pressed={isActive("bulletList")} data-active={isActive("bulletList") || undefined} onClick={() => editor?.chain().focus().toggleBulletList().run()}><List size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Ordered list" aria-pressed={isActive("orderedList")} data-active={isActive("orderedList") || undefined} onClick={() => editor?.chain().focus().toggleOrderedList().run()}><ListOrdered size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Blockquote" aria-pressed={isActive("blockquote")} data-active={isActive("blockquote") || undefined} onClick={() => editor?.chain().focus().toggleBlockquote().run()}><Quote size={16} /></Button>
          <Button variant="ghost" size="sm" aria-label="Horizontal rule" onClick={() => editor?.chain().focus().setHorizontalRule().run()}><Minus size={16} /></Button>
          <Button
            ref={linkBtnRef}
            variant="ghost"
            size="sm"
            aria-label="Insert link"
            aria-pressed={isActive("link")}
            aria-expanded={linkOpen}
            data-active={isActive("link") || undefined}
            onClick={() => {
              const prev = (editor?.getAttributes("link") as { href?: string } | undefined)?.href ?? "";
              setLinkUrl(prev);
              setLinkOpen(o => !o);
            }}
          >
            <Link2 size={16} />
          </Button>
        </div>
      );

      const linkPopover = linkOpen && linkBtnRef.current ? (
        <DSPortal>
          <div
            className="ds-atom-richtext-linkpopover"
            role="dialog"
            aria-modal="false"
            aria-label="Edit link URL"
            style={{
              position: "fixed",
              top: linkBtnRef.current.getBoundingClientRect().bottom + 4,
              left: linkBtnRef.current.getBoundingClientRect().left,
            }}
          >
            <input
              type="url"
              value={linkUrl}
              onChange={(e) => setLinkUrl(e.target.value)}
              placeholder="https://example.com"
              className="ds-atom-richtext-linkinput"
              onKeyDown={(e) => {
                if (e.key === "Escape") { e.preventDefault(); setLinkOpen(false); }
                else if (e.key === "Enter") {
                  e.preventDefault();
                  if (linkUrl) editor?.chain().focus().extendMarkRange("link").setLink({ href: linkUrl }).run();
                  else editor?.chain().focus().unsetLink().run();
                  setLinkOpen(false);
                }
              }}
              autoFocus
            />
            <div className="ds-atom-richtext-linkactions">
              <Button variant="ghost" size="sm" onClick={() => { editor?.chain().focus().unsetLink().run(); setLinkOpen(false); }}>Remove</Button>
              <Button variant="primary" size="sm" onClick={() => {
                if (linkUrl) editor?.chain().focus().extendMarkRange("link").setLink({ href: linkUrl }).run();
                setLinkOpen(false);
              }}>Apply</Button>
            </div>
          </div>
        </DSPortal>
      ) : null;

      if (!editor) {
        return <div ref={ref} className={`ds-atom-richtext${className ? ` ${className}` : ""}`} style={style} aria-label={ariaLabel} aria-busy="true" />;
      }

      return (
        <div
          ref={ref}
          className={`ds-atom-richtext${className ? ` ${className}` : ""}`}
          style={style}
          aria-label={ariaLabel}
        >
          {!readOnly && (toolbar ?? defaultToolbar)}
          <div className="ds-atom-richtext-surface">
            <EditorContent editor={editor} />
          </div>
          {linkPopover}
        </div>
      );
    });
    ```
