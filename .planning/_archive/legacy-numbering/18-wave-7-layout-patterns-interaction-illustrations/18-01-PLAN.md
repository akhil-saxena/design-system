    Create `src/AppShell.tsx`:

    ```typescript
    import { type ReactElement, type ReactNode, forwardRef, useEffect, useState } from "react";

    export interface AppShellProps {
      /** Sidebar nav component — receives collapsed + onToggleCollapse via cloneElement */
      sidebar: ReactElement<{ collapsed?: boolean; onToggleCollapse?: () => void }>;
      /** Topbar component (AppBar DS-72 or any ReactNode) */
      topbar: ReactNode;
      /** Main page content */
      main: ReactNode;
      /** Optional footer (DS-73 or any ReactNode) */
      footer?: ReactNode;
      /**
       * localStorage key for sidebar collapse persistence.
       * Pass null to disable persistence.
       * @default "ds-sidebar-collapsed"
       */
      storageKey?: string | null;
      /**
       * Expanded sidebar width in pixels.
       * @default 240
       */
      sidebarWidth?: number;
      className?: string;
      style?: React.CSSProperties;
    }
    ```

    Implement with:
    - `useState<boolean>` initializer reads `window.localStorage.getItem(storageKey) === 'true'` guarded by `typeof window !== 'undefined' && storageKey !== null`
    - `useEffect` writes `window.localStorage.setItem(storageKey, String(collapsed))` guarded by same check; deps: `[collapsed, storageKey]`
    - Root `<div className="ds-atom-appshell" data-sidebar-collapsed={collapsed}` with `style={{ '--ds-sidebar-w': collapsed ? '48px' : `${sidebarWidth}px` } as React.CSSProperties}`
    - `<header className="ds-atom-appshell-topbar">{topbar}</header>`
    - `<aside className="ds-atom-appshell-sidebar">{React.cloneElement(sidebar, { collapsed, onToggleCollapse: () => setCollapsed(c => !c) })}</aside>`
    - `<main className="ds-atom-appshell-main">{main}</main>`
    - `{footer && <footer className="ds-atom-appshell-footer">{footer}</footer>}`

    Import `React` for `cloneElement`. Use `forwardRef<HTMLDivElement, AppShellProps>`.
