    Create `src/Wizard.tsx`:

    ```typescript
    import { type ReactNode, forwardRef, useState, useRef } from "react";
    import { ProgressBar } from "./ProgressBar";
    import { useFocusTrap } from "./hooks/useFocusTrap";

    export interface WizardStep {
      label: string;
      desc?: string;
      /** Return error string to block advance; return null/undefined to proceed */
      validate?: () => string | null | undefined;
    }

    export interface WizardProps {
      steps: WizardStep[];
      onComplete: () => void;
      onCancel?: () => void;
      orientation?: "horizontal" | "vertical";  // default "horizontal"
      children: ReactNode | ((step: number) => ReactNode);
    }
    ```

    State:
    - `const [current, setCurrent] = useState(0)`
    - `const [error, setError] = useState<string | null>(null)`
    - `const [trapEl, setTrapEl] = useState<HTMLDivElement | null>(null)`
    - `useFocusTrap(trapEl, true)` — trap is always active (Wizard is always interactive)

    `handleNext`:
    1. Run `steps[current].validate?.()` — if returns string, `setError(string)` and return
    2. Clear error: `setError(null)`
    3. If `current === steps.length - 1`: call `onComplete()`; return
    4. Else: `setCurrent(c => c + 1)`

    `handleBack`:
    1. If `current === 0` return
    2. `setError(null)`
    3. `setCurrent(c => c - 1)`

    Structure:
    ```tsx
    <div
      ref={setTrapEl}  // callback ref for useFocusTrap
      className="ds-atom-wizard"
      data-orientation={orientation ?? "horizontal"}
    >
      {/* Stepper */}
      <div className="ds-atom-wizard-stepper">
        {steps.map((step, i) => {
          const status = i < current ? "done" : i === current ? "active" : "pending";
          return (
            <Fragment key={i}>
              {i > 0 && <div className="ds-atom-wizard-connector" aria-hidden="true" />}
              <div className="ds-atom-wizard-step" data-status={status}>
                <div className="ds-atom-wizard-dot">
                  {status === "done" ? <CheckSVG /> : i + 1}
                </div>
                <div className="ds-atom-wizard-label">{step.label}</div>
                {step.desc && <div className="ds-atom-wizard-desc">{step.desc}</div>}
              </div>
            </Fragment>
          );
        })}
      </div>

      {/* Progress bar */}
      <ProgressBar value={(current / steps.length) * 100} />

      {/* Step content */}
      <div className="ds-atom-wizard-content">
        {typeof children === "function" ? children(current) : children}
      </div>

      {/* Validation error */}
      {error && <div className="ds-atom-wizard-error" role="alert">{error}</div>}

      {/* Navigation — INSIDE the trap boundary */}
      <div className="ds-atom-wizard-nav">
        {onCancel && <button type="button" className="ds-atom-wizard-cancel" onClick={onCancel}>Cancel</button>}
        <button type="button" className="ds-atom-wizard-back" onClick={handleBack} disabled={current === 0}>Back</button>
        <button type="button" className="ds-atom-wizard-next" onClick={handleNext}>
          {current === steps.length - 1 ? "Finish" : "Next"}
        </button>
      </div>
    </div>
    ```

    The inline check SVG for "done" steps (to avoid icon import complexity):
    ```tsx
    const CheckSVG = () => (
      <svg viewBox="0 0 24 24" width="13" height="13" fill="none" stroke="var(--cream)" strokeWidth="3" aria-hidden="true">
        <polyline points="20 6 9 17 4 12" />
      </svg>
    );
    ```
