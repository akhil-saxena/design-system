    ```typescript
    type InlineEditState = "idle" | "editing" | "saving" | "error";

    export interface InlineEditProps {
      value: string;
      onSave: (value: string) => Promise<void> | void;
      multiline?: boolean;       // false=input, true=textarea
      placeholder?: string;
      disabled?: boolean;
      className?: string;
      style?: CSSProperties;
    }
    ```

    State:
    - `const [state, setState] = useState<InlineEditState>("idle")`
    - `const [draft, setDraft] = useState(value)` — reset to value when idle
    - `const [errorMsg, setErrorMsg] = useState<string | null>(null)`
    - Sync draft with external value changes: `useEffect(() => { if (state === "idle") setDraft(value); }, [value, state])`

    `handleCommit`:
    1. `setState("saving")`
    2. Try: `await onSave(draft)` → `setState("idle")`
    3. Catch error: `setState("error")`, `setErrorMsg(error?.message ?? "Save failed")`

    `handleCancel`:
    1. `setDraft(value)` (restore original)
    2. `setState("idle")`
    3. `setErrorMsg(null)`

    Keyboard handler on input/textarea:
    - `Enter` (not Shift+Enter for textarea): `handleCommit()`
    - `Escape`: `handleCancel()`

    Blur handler: `handleCancel()` — clicking outside cancels

    Idle state renders:
    ```tsx
    <span
      className="ds-atom-inlineedit"
      data-state="idle"
      onClick={disabled ? undefined : () => setState("editing")}
      tabIndex={disabled ? -1 : 0}
      onKeyDown={e => { if (e.key === "Enter" || e.key === " ") { e.preventDefault(); if (!disabled) setState("editing"); } }}
      role="button"
      aria-label="Click to edit"
    >
      {value || <span style={{ color: "var(--ink-4)" }}>{placeholder}</span>}
    </span>
    ```

    Editing/saving/error state renders input or textarea.

    Create `src/SearchAndFilters.tsx`:

    ```typescript
    import { type ReactNode, useRef, useState } from "react";
    import { DSDropdown } from "./_internals/DSDropdown";
    import { Chip } from "./Chip";

    export interface SearchFilter {
      id: string;
      label: string;
      tone?: "default" | "match" | "miss" | "learning" | "tag";
    }

    export interface SearchSuggestion {
      id: string;
      label: string;
    }

    export interface SearchAndFiltersProps {
      value?: string;
      placeholder?: string;                    // default "Search..."
      suggestions?: SearchSuggestion[];        // shown in DSDropdown when input focused + has value
      activeFilters?: SearchFilter[];          // filter chips shown below input
      onSearch?: (value: string) => void;
      onSuggestionSelect?: (suggestion: SearchSuggestion) => void;
      onFilterRemove?: (filter: SearchFilter) => void;
      onClearFilters?: () => void;
      className?: string;
      style?: React.CSSProperties;
    }
    ```

    Implementation:
    - `const inputRef = useRef<HTMLInputElement>(null)`
    - `const [open, setOpen] = useState(false)`
    - `const [activeIndex, setActiveIndex] = useState(-1)`
    - `const [inputValue, setInputValue] = useState(value ?? "")`
    - Open dropdown on focus when suggestions exist; close on blur (via onOpenChange)
    - On suggestion select: `onSuggestionSelect?.(suggestions[idx])`, close dropdown, clear input

    Structure:
    ```tsx
    <div className="ds-atom-searchfilters" style={style}>
      <div className="ds-atom-searchfilters-bar">
        <input
          ref={inputRef}
          type="search"
          className="ds-atom-searchfilters-input"
          placeholder={placeholder ?? "Search..."}
          value={inputValue}
          onChange={e => { setInputValue(e.target.value); onSearch?.(e.target.value); if (suggestions?.length) setOpen(true); }}
          onFocus={() => { if (suggestions?.length && inputValue) setOpen(true); }}
          aria-expanded={open}
          aria-haspopup="listbox"
        />
        {activeFilters && activeFilters.length > 0 && (
          <button type="button" className="ds-atom-searchfilters-clear" onClick={onClearFilters}>
            Clear all
          </button>
        )}
      </div>

      {suggestions && suggestions.length > 0 && (
        <DSDropdown
          anchorRef={inputRef}
          open={open}
          onOpenChange={setOpen}
          activeIndex={activeIndex}
          onActiveIndexChange={setActiveIndex}
          itemCount={suggestions.length}
          onSelect={idx => { onSuggestionSelect?.(suggestions[idx]!); setOpen(false); setInputValue(""); }}
          typeAheadGetText={idx => suggestions[idx]?.label ?? ""}
        >
          <ul role="listbox" style={{ listStyle: "none", margin: 0, padding: 0 }}>
            {suggestions.map((s, i) => (
              <li
                key={s.id}
                role="option"
                aria-selected={i === activeIndex}
                className="ds-atom-searchfilters-option"
                data-active={i === activeIndex}
              >
                {s.label}
              </li>
            ))}
          </ul>
        </DSDropdown>
      )}

      {activeFilters && activeFilters.length > 0 && (
        <div className="ds-atom-searchfilters-chips">
          {activeFilters.map(f => (
            <Chip
              key={f.id}
              tone={f.tone ?? "default"}
              onRemove={() => onFilterRemove?.(f)}
            >
              {f.label}
            </Chip>
          ))}
        </div>
      )}
    </div>
    ```
