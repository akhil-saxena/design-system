  <action>
    Create `src/Coachmark.tsx`:

    ```typescript
    import { type ReactNode, type RefObject, useRef, useState } from "react";
    import { Popover } from "./Popover";

    export interface CoachmarkProps {
      /** Ref to the element being highlighted */
      anchorRef: RefObject<HTMLElement | null>;
      title: string;
      desc?: string;
      /** Current step number (1-based) */
      step?: number;
      /** Total steps in the coachmark sequence */
      total?: number;
      onNext?: () => void;
      onDone?: () => void;
      /**
       * localStorage key for persist-dismiss state.
       * Omit or pass null to disable persistence.
       */
      storageKey?: string | null;
      placement?: "bottom-start" | "bottom-end" | "top-start" | "top-end";
    }
    ```

    Implementation:
    - `useState` initializer checks localStorage: `typeof window !== 'undefined' && storageKey ? localStorage.getItem(storageKey) === 'dismissed' : false`
    - `dismiss()`: sets dismissed=true; if storageKey: `localStorage.setItem(storageKey, 'dismissed')`
    - Renders `<Popover anchorRef={anchorRef} open={!dismissed} onOpenChange={(o) => { if (!o) dismiss(); }} placement={placement ?? "bottom-start"}>`
    - Inside Popover: `<div className="ds-atom-coachmark">`
      - Step indicator: `{step && total && <span className="ds-atom-coachmark-step">{step} / {total}</span>}`
      - Title + desc
      - Nav dot row: `{total && Array.from({length: total}, (_, i) => <span key={i} className="ds-atom-coachmark-dot" data-active={i === (step ?? 1) - 1} />)}`
      - Button row: `{step < total ? <button onClick={onNext}>Next</button> : <button onClick={onDone}>Done</button>}` + `<button onClick={dismiss} aria-label="Dismiss">×</button>`
